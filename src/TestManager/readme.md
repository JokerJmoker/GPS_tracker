// ============================================
// ОБЩИЙ ФОРМАТ NMEA СООБЩЕНИЙ
// ============================================

// СТРУКТУРА СТАНДАРТНОГО NMEA ПРЕДЛОЖЕНИЯ:
// $XXYYY,data1,data2,data3,...,dataN*CS
// |  |    |                    |     |
// |  |    |                    |     +-- Контрольная сумма (hex)
// |  |    |                    +-------- Разделители полей (запятые)
// |  |    +----------------------------- Данные (поля)
// |  +---------------------------------- Идентификатор сообщения (3 буквы)
// +------------------------------------- Идентификатор системы (2 буквы)
//                                       (GP=GPS, GL=GLONASS, BD=BeiDou, GA=Galileo, GN=комбинированный)

// ============================================
// ТИПЫ СИСТЕМ (первые 2 буквы)
// ============================================
// GP - GPS (США)
// GL - GLONASS (Россия)
// GA - Galileo (Европа)
// BD - BeiDou (Китай)
// GN - Комбинированный (две и более систем)
// QZ - QZSS (Япония)

// ============================================
// ОСНОВНЫЕ ТИПЫ СООБЩЕНИЙ (последние 3 буквы)
// ============================================

// 1. GGA - Global Positioning System Fix Data (данные о позиции и качестве)
// Формат: $xxGGA,time,lat,NS,lon,EW,quality,numSV,hdop,alt,altUnit,sep,sepUnit,diffAge,diffStation*cs
// Поля:
//   time      - UTC время (hhmmss.ss)
//   lat       - Широта (ddmm.mmmm)
//   NS        - N=север, S=юг
//   lon       - Долгота (dddmm.mmmm)
//   EW        - E=восток, W=запад
//   quality   - Качество: 0=нет fix, 1=GPS fix, 2=DGPS, 3=PPS, 4=RTK, 5=float RTK
//   numSV     - Количество используемых спутников
//   hdop      - Horizontal Dilution of Precision (1.0 = отлично, >2.0 = плохо)
//   alt       - Высота над уровнем моря
//   altUnit   - Единица высоты (M=метры)
//   sep       - Высота геоида над эллипсоидом
//   sepUnit   - Единица (M)
//   diffAge   - Возраст дифференциальной поправки
//   diffStation - ID дифференциальной станции

// 2. RMC - Recommended Minimum Specific GNSS Data (рекомендованный минимум)
// Формат: $xxRMC,time,status,lat,NS,lon,EW,speed,course,date,mv,mvEW,posMode*cs
// Поля:
//   time    - UTC время (hhmmss.ss)
//   status  - A=valid (активен), V=invalid (недействителен)
//   lat     - Широта (ddmm.mmmm)
//   NS      - N/S
//   lon     - Долгота (dddmm.mmmm)
//   EW      - E/W
//   speed   - Скорость над землей (узлы)
//   course  - Курс (градусы)
//   date    - Дата (ddmmyy)
//   mv      - Магнитное склонение
//   mvEW    - E/W магнитного склонения
//   posMode - Режим: A=автономный, D=дифференциальный, E=оценочный, M=ручной

// 3. GSA - GNSS DOP and Active Satellites (DOP и активные спутники)
// Формат: $xxGSA,opMode,navMode,sat1,sat2,...,sat12,PDOP,HDOP,VDOP*cs
// Поля:
//   opMode  - M=manual, A=auto
//   navMode - 1=fix нет, 2=2D fix, 3=3D fix
//   sat1-12 - ID используемых спутников (до 12)
//   PDOP    - Positional DOP (общая точность)
//   HDOP    - Horizontal DOP (горизонтальная)
//   VDOP    - Vertical DOP (вертикальная)

// 4. GSV - Satellites in View (видимые спутники)
// Формат: $xxGSV,totalMsgs,msgNum,numSV,sat1,el1,az1,cno1,sat2,el2,az2,cno2,..*cs
// Поля:
//   totalMsgs - Всего сообщений (1-3)
//   msgNum    - Номер текущего сообщения
//   numSV     - Всего видимых спутников
//   satN      - ID спутника (1-255)
//   elN       - Угол места (0-90 градусов)
//   azN       - Азимут (0-359 градусов)
//   cnoN      - Сигнал/шум (0-99 dBHz, 0=нет сигнала)
// Примечание: В одном сообщении до 4 спутников

// 5. VTG - Course Over Ground and Ground Speed (курс и скорость)
// Формат: $xxVTG,courseT,T,courseM,M,speedN,N,speedK,K,posMode*cs
// Поля:
//   courseT - Курс истинный (градусы)
//   T       - True (индикатор)
//   courseM - Курс магнитный (градусы)
//   M       - Magnetic (индикатор)
//   speedN  - Скорость в узлах
//   N       - Knots (индикатор)
//   speedK  - Скорость в км/ч
//   K       - Kilometers (индикатор)
//   posMode - Режим позиционирования

// 6. GLL - Geographic Position (широта/долгота)
// Формат: $xxGLL,lat,NS,lon,EW,time,status,posMode*cs
// Поля:
//   lat     - Широта
//   NS      - N/S
//   lon     - Долгота
//   EW      - E/W
//   time    - UTC время
//   status  - A=valid, V=invalid
//   posMode - Режим позиционирования

// 7. ZDA - Time and Date (дата и время)
// Формат: $xxZDA,time,day,month,year,localZone,zoneMinutes*cs
// Поля:
//   time       - UTC время (hhmmss.ss)
//   day        - День (01-31)
//   month      - Месяц (01-12)
//   year       - Год
//   localZone  - Смещение локального часового пояса (00-12)
//   zoneMinutes - Минуты смещения (00-59)

// 8. TXT - Text Message (текстовое сообщение)
// Формат: $xxTXT,code1,code2,code3,textMessage*cs
// Поля:
//   code1,code2,code3 - Коды сообщения
//   textMessage      - Текстовое сообщение (например "ANTENNA OK")

// ============================================
// ПОКАЗАТЕЛИ КАЧЕСТВА (DOP)
// ============================================
// PDOP (Positional) - <1=идеально, 1-2=отлично, 2-5=хорошо, >5=плохо
// HDOP (Horizontal) - <1=идеально, 1-2=отлично, 2-5=приемлемо, >5=плохо
// VDOP (Vertical)   - <1=идеально, 1-2=отлично, 2-5=приемлемо, >5=плохо

// ============================================
// УРОВНИ СИГНАЛА (C/No)
// ============================================
// 0-20  - Очень слабый (нет фикса)
// 20-30 - Слабый (сложно фиксировать)
// 30-40 - Хороший (стабильный фикс)
// 40-50 - Отличный
// 50+   - Идеальный

// ============================================
// КАЧЕСТВО FIX'А (GGA quality)
// ============================================
// 0 - Нет fix'а
// 1 - GPS fix (обычный режим)
// 2 - DGPS fix (дифференциальный)
// 3 - PPS fix (точный)
// 4 - RTK fix (Real Time Kinematic - сантиметры)
// 5 - Float RTK (плавающий RTK)
// 6 - Estimated (оценочный)

// ============================================
// СТАТУСЫ (RMC status)
// ============================================
// A - Active (данные действительны)
// V - Void (данные недействительны)

// ============================================
// КОНТРОЛЬНАЯ СУММА
// ============================================
// Алгоритм: XOR всех байтов между $ и *
// Пример расчета:
// Исходное: $GPGGA,123519,4807.038,N,01131.000,E,1,08,0.9,545.4,M,46.9,M,,*47
// Контрольная сумма: XOR('G','P','G','G','A',',','1','2','3','5','1','9', ... ) = 0x47
// Формула в коде: uint8_t cs = 0; while(*s) cs ^= *s++;
